SELECT Category.Name, (SUM(Product.Quantity)/Sale.Quantity)*100&'%'
FROM Category, Product, Sale
WHERE Category.IdCategory=Product.IdCategory AND Product.IdProduct=Sale.IdProduct;

/* використовуючи підзапити вивести коди та назви всіх товарів які поставляються
лише визначеним постачальником*/

SELECT Product.IdProduct, Product.Name
FROM Product, Delivery, Supplier 
WHERE Product.IdProduct=Delivery.IdProduct AND 
      Delivery.IdSupplier=Supplier.IdSupplier AND
      Supplier.Name=''
GROUP BY Product.IdProduct, Product.Name;
